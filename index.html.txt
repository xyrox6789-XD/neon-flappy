                Score: <span id="finalScore" style="color: #fff; font-weight: bold;">0</span> | 
                Best: <span id="highScore" style="color: #ffae00; font-weight: bold;">0</span>

<script>
const canvas = document.getElementById('gameCanvas');
const ctx = canvas.getContext('2d');
const hud = document.getElementById('hud');
const uiLayer = document.getElementById('uiLayer');
const menuCard = document.getElementById('menuCard');
const menuTitle = document.getElementById('menuTitle');
const menuSub = document.getElementById('menuSub');
const startBtn = document.getElementById('startBtn');
const statsDiv = document.getElementById('stats');
const finalScoreSpan = document.getElementById('finalScore');
const highScoreSpan = document.getElementById('highScore');

// Game States
let gameState = 'START'; // START, PLAYING, GAMEOVER
let score = 0;
let highscore = localStorage.getItem('flappy_highscore') || 0;

// Environment / Physics config
const GRAVITY = 0.25;
const FLAP_STRENGTH = -5.5;
const MAX_FALL_SPEED = 8;
const PIPE_SPEED = 2.5;
const PIPE_SPAWN_RATE = 100; // frames
const GAP_HEIGHT = 160;

let frameCount = 0;
let screenShake = 0;

// Game Entities
let bird;
let pipes = [];
let particles = [];
let stars = []; // Background stars/particles

class Bird {
    constructor() {
        this.x = 100;
        this.y = 300;
        this.radius = 14;
        this.velocity = 0;
        this.angle = 0;
    }

    update() {
        this.velocity += GRAVITY;
        if (this.velocity > MAX_FALL_SPEED) this.velocity = MAX_FALL_SPEED;
        this.y += this.velocity;

        // Dynamic rotation based on velocity
        if (this.velocity < 0) {
            this.angle = Math.max(-0.4, this.angle - 0.1);
        } else if (this.velocity > 2) {
            this.angle = Math.min(0.7, this.angle + 0.05);
        }

        // Floor / Ceiling bounds
        if (this.y + this.radius > canvas.height) {
            this.y = canvas.height - this.radius;
            triggerGameOver();
        }
        if (this.y - this.radius < 0) {
            this.y = this.radius;
            this.velocity = 0;
        }
    }

    flap() {
        this.velocity = FLAP_STRENGTH;
        // Spawn energetic thruster particles behind the bird
        for(let i=0; i<5; i++) {
            particles.push(new Particle(this.x - 10, this.y, '#00b4db'));
        }
    }

    draw() {
        ctx.save();
        ctx.translate(this.x, this.y);
        ctx.rotate(this.angle);

        // Draw sleek neon bird
        let grad = ctx.createRadialGradient(0, 0, 2, 0, 0, this.radius);
        grad.addColorStop(0, '#fff');
        grad.addColorStop(0.4, '#00b4db');
        grad.addColorStop(1, '#002f4b');
        
        ctx.beginPath();
        ctx.arc(0, 0, this.radius, 0, Math.PI * 2);
        ctx.fillStyle = grad;
        ctx.shadowBlur = 15;
        ctx.shadowColor = '#00b4db';
        ctx.fill();
        ctx.closePath();

        // Eye
        ctx.beginPath();
        ctx.arc(6, -4, 3, 0, Math.PI * 2);
        ctx.fillStyle = '#fff';
        ctx.fill();
        
        ctx.restore();
        ctx.shadowBlur = 0; // reset
    }
}

class Pipe {
    constructor() {
        this.x = canvas.width;
        this.width = 70;
        // Ensure pipes aren't too high or low
        this.topHeight = Math.floor(Math.random() * (canvas.height - GAP_HEIGHT - 120)) + 60;
        this.bottomY = this.topHeight + GAP_HEIGHT;
        this.passed = false;
    }

    update() {
        this.x -= PIPE_SPEED;
    }

    draw() {
        ctx.save();
        ctx.shadowBlur = 10;
        ctx.shadowColor = '#ff5e62';

        // Glowing Neon Gradients for Pipes
        let topGrad = ctx.createLinearGradient(this.x, 0, this.x + this.width, 0);
        topGrad.addColorStop(0, '#ff5e62');
        topGrad.addColorStop(1, '#ff9966');

        // Top Pipe
        ctx.fillStyle = topGrad;
        ctx.fillRect(this.x, 0, this.width, this.topHeight);
        // Lip of top pipe
        ctx.fillRect(this.x - 4, this.topHeight - 20, this.width + 8, 20);

        // Bottom Pipe
        ctx.fillRect(this.x, this.bottomY, this.width, canvas.height - this.bottomY);
        // Lip of bottom pipe
        ctx.fillRect(this.x - 4, this.bottomY, this.width + 8, 20);

        ctx.restore();
        ctx.shadowBlur = 0;
    }

    collidesWith(b) {
        if (b.x + b.radius > this.x && b.x - b.radius < this.x + this.width) {
            if (b.y - b.radius < this.topHeight || b.y + b.radius > this.bottomY) {
                return true;
            }
        }
        return false;
    }
}

class Particle {
    constructor(x, y, color) {
        this.x = x;
        this.y = y;
        this.vx = (Math.random() - 0.5) * 4 - 2;
        this.vy = (Math.random() - 0.5) * 4;
        this.radius = Math.random() * 3 + 1;
        this.alpha = 1;
        this.color = color || '#fff';
    }
    update() {
        this.x += this.vx;
        this.y += this.vy;
        this.alpha -= 0.02;
    }
    draw() {
        ctx.save();
        ctx.globalAlpha = this.alpha;
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI*2);
        ctx.fillStyle = this.color;
        ctx.fill();
        ctx.restore();
    }
}

// Background space dust
function initStars() {
    stars = [];
    for(let i=0; i<40; i++) {
        stars.push({
            x: Math.random() * canvas.width,
            y: Math.random() * canvas.height,
            size: Math.random() * 2,
            speed: Math.random() * 0.5 + 0.1
        });
    }
}

function drawBackground() {
    // Star field movement
    ctx.fillStyle = 'rgba(255,255,255,0.3)';
    stars.forEach(star => {
        ctx.fillRect(star.x, star.y, star.size, star.size);
        if (gameState === 'PLAYING') {
            star.x -= star.speed;
            if (star.x < 0) star.x = canvas.width;
        }
    });

    // Elegant backdrop landscape hints
    ctx.fillStyle = 'rgba(255, 255, 255, 0.03)';
    ctx.beginPath();
    ctx.moveTo(0, canvas.height);
    ctx.lineTo(0, canvas.height - 150);
    ctx.quadraticCurveTo(canvas.width/2, canvas.height - 200, canvas.width, canvas.height - 130);
    ctx.lineTo(canvas.width, canvas.height);
    ctx.fill();
}

function resetGame() {
    bird = new Bird();
    pipes = [];
    particles = [];
    score = 0;
    frameCount = 0;
    hud.innerText = score;
}

function startGame() {
    resetGame();
    gameState = 'PLAYING';
    uiLayer.style.display = 'none';
    hud.style.display = 'block';
}

function triggerGameOver() {
    if (gameState === 'GAMEOVER') return;
    gameState = 'GAMEOVER';
    screenShake = 20;

    // Crash explosion particles
    for(let i=0; i<30; i++) {
        particles.push(new Particle(bird.x, bird.y, '#ff5e62'));
    }

    if(score > highscore) {
        highscore = score;
        localStorage.setItem('flappy_highscore', highscore);
    }

    // Update UI elements
    menuTitle.innerText = "GAME OVER";
    menuSub.innerText = "The grid claimed you.";
    finalScoreSpan.innerText = score;
    highScoreSpan.innerText = highscore;
    statsDiv.style.display = 'block';
    startBtn.innerText = 'TRY AGAIN';
    
    setTimeout(() => {
        uiLayer.style.display = 'flex';
        hud.style.display = 'none';
    }, 500);
}

// Global Actions
function handleAction() {
    if (gameState === 'PLAYING') {
        bird.flap();
    } else if (gameState === 'START') {
        startGame();
    }
}

// Input Event Listeners
window.addEventListener('keydown', (e) => {
    if (e.code === 'Space') {
        e.preventDefault();
        handleAction();
    }
});
canvas.addEventListener('touchstart', (e) => {
    e.preventDefault();
    handleAction();
});
canvas.addEventListener('mousedown', (e) => {
    if (gameState === 'PLAYING') handleAction();
});
startBtn.addEventListener('click', (e) => {
    e.stopPropagation(); // prevent double triggering
    startGame();
});

// Engine Loop
function loop() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);

    // Apply Screen Shake if active
    ctx.save();
    if (screenShake > 0) {
        let dx = (Math.random() - 0.5) * screenShake;
        let dy = (Math.random() - 0.5) * screenShake;
        ctx.translate(dx, dy);
        screenShake *= 0.9; // Decay
        if (screenShake < 0.5) screenShake = 0;
    }

    drawBackground();

    if (gameState === 'PLAYING') {
        frameCount++;

        // Spawn Pipes
        if (frameCount % PIPE_SPAWN_RATE === 0) {
            pipes.push(new Pipe());
        }

        bird.update();

        // Pipes engine
        for (let i = pipes.length - 1; i >= 0; i--) {
            pipes[i].update();
            pipes[i].draw();

            if (pipes[i].collidesWith(bird)) {
                triggerGameOver();
            }

            // Earn point
            if (!pipes[i].passed && pipes[i].x + pipes[i].width < bird.x) {
                score++;
                hud.innerText = score;
                pipes[i].passed = true;
                
                // Score pop particles
                for(let p=0; p<8; p++) {
                    particles.push(new Particle(bird.x + 20, bird.y, '#ffae00'));
                }
            }

            // Remove out-of-screen pipes
            if (pipes[i].x + pipes[i].width < 0) {
                pipes.splice(i, 1);
            }
        }
    } else {
        // Draw inactive elements for visual aesthetics behind menus
        pipes.forEach(pipe => pipe.draw());
    }

    // Bird rendering
    if (gameState !== 'GAMEOVER') {
        bird.draw();
    }

    // Particles handler
    for (let i = particles.length - 1; i >= 0; i--) {
        particles[i].update();
        particles[i].draw();
        if (particles[i].alpha <= 0) {
            particles.splice(i, 1);
        }
    }

    ctx.restore();
    requestAnimationFrame(loop);
}

// Run Setup
initStars();
bird = new Bird(); // Dummy bird for start menu aesthetics
loop();

</script>
